    // Lightweight AR/VR mode toggles (no external libs)

    (function() {

      const stage = document.getElementById('immersiveStage');

      const badge = document.getElementById('modeBadge');

      const orbitBtn = document.getElementById('modeOrbit');

      const arBtn = document.getElementById('modeAR');

      const vrBtn = document.getElementById('modeVR');


      function setActive(btn) {

        [orbitBtn, arBtn, vrBtn].forEach(b => {

          const active = b === btn;

          b.classList.toggle('active', active);

          b.setAttribute('aria-pressed', String(active));

        });

      }


      orbitBtn.addEventListener('click', () => {

        stage.classList.remove('ar','vr'); stage.classList.add('orbit');

        badge.textContent = 'Orbit Preview'; setActive(orbitBtn);

      });

      arBtn.addEventListener('click', () => {

        stage.classList.remove('orbit','vr'); stage.classList.add('ar');

        badge.textContent = 'AR Preview'; setActive(arBtn);

      });

      vrBtn.addEventListener('click', () => {

        stage.classList.remove('orbit','ar'); stage.classList.add('vr');

        badge.textContent = 'VR Preview'; setActive(vrBtn);

      });

    })();

  </script>
